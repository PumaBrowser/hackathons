transactions¶

select transactions.transaction_hash, transaction_actions.action_kind, transaction_actions.args, transactions.included_in_block_hash, transactions.block_timestamp, transactions.signer_account_id, transactions.signer_public_key, transactions.receiver_account_id, transactions.converted_into_receipt_id from transaction_actions inner join transactions on transaction_actions.transaction_hash = transactions.transaction_hash limit 10

receipts¶

select receipts.receipt_id, receipts.included_in_block_hash, receipts.included_in_block_timestamp, receipts.predecessor_account_id, receipts.receiver_account_id, receipts.originated_from_transaction_hash, action_receipt_actions.action_kind, action_receipt_actions.args, action_receipt_actions.receipt_receiver_account_id, action_receipt_actions.receipt_included_in_block_timestamp from action_receipt_actions inner join receipts on receipts.receipt_id = action_receipt_actions.receipt_id limit 10

execution¶

select * from execution_outcomes inner join execution_outcome_receipts on execution_outcomes.receipt_id = execution_outcome_receipts.executed_receipt_id

accounts¶

select * from account_changes inner join accounts on accounts.created_by_receipt_id = account_changes.caused_by_receipt_id
